Was ist Angular?

Angular ist ein auf TypeScript basierendes JavaScript Framework, welches dir erlaubt Single-Page-Webanwendung zu erstellen.

Wie funktioniert Angular?

AppComponent


                            import {Component, OnInit} from '@angular/core';
                            import {SomeService} from './some.service';
                        

                            @Component({
                                selector: 'app-root',
                                templateUrl: './app.component.html',
                                styleUrls: ['./app.component.css']
                            })
                        

                            export class AppComponent implements OnInit {
                                title: String;

                                constructor(private someService: SomeService) {
                                }

                                ngOnInit() {
                                    this.title = this.someService.getTitle();
                                }
                            }

Databinding

String Interpolation


                            <h1>
                                {{ 'Welcome' }} to App!
                            </h1>
                        

                            <h1>
                                {{ 12 }} to App!
                            </h1>
                        

                            <h1>
                                Welcome to {{ title }}!
                            </h1>
                        

                            <h1>
                                Welcome to {{ someService.getTitle() }}!
                            </h1>

Property Binding


                            <input [placeholder]="title" />
                        

                            <img [src]="imageUrl">
                        

                            <button [disabled]="isUnchanged">Cancel is disabled </button>

Event Binding


                            <button (click)="event()">
                                Hochzählen
                            </button>
                        

                            <button (click)="buttonClicks = buttonClicks + 1">
                                Hochzählen
                            </button>
                        

                            <input (input)="title = $event.target.value" />
                        

                            <input (input)="title = function($event)" />

Two-Way-Databinding


                            <input [(ngModel)]="title" />

ist äquivalent zu


                            <input [value]="title" (input)="title"/>

Directives

if


                            <p *ngIf="true">If</p>
                        

                            <p *ngIf="title !== 'app'">If</p>
                        

                            <p *ngIf="isUpdated()">If</p>

if-else


                            <p *ngIf="!(title === 'app'); else ifelse">If</p>
                            <ng-template #ifelse>
                                <p>Else</p>
                            </ng-template>

for


                              <p *ngFor="let number of numbers">{{ number }}</p>
                        

                              <p *ngFor="let number of numbers; let i = index"></p>
                        

                              <app-number
                                      *ngFor="let nr of numbers; let i = index"
                                      [number]="nr"
                                      [index]="i"></app-number>

ngClass


                            .something {
                                background-color: blue;
                                color: white;
                            }
                        

                            <p [ngClass]="{ something: title !== 'app' }">If</p>

ngStyle


                            <p [ngStyle]="{backgroundColor: 'red'}">Else</p>
                        

                            <p [ngStyle]="{'background-color': 'red'}">Else</p>
                        

                            <p [ngStyle]="{backgroundColor: getColor()}">Else</p>
